import java.util.Scanner;

public class Factorial {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter a number: ");
        int N = sc.nextInt();

        long fact = 1;
        for (int i = 1; i <= N; i++) {
            fact *= i;
        }

        System.out.println("Factorial of " + N + " is: " + fact);
        sc.close();
    }
}
